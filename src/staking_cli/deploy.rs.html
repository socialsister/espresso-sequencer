61    pub async fn deploy_version(
62        stake_table_contract_version: StakeTableContractVersion,
63    ) -> Result<Self> {
64        let exit_escrow_period = Duration::from_secs(1);
65        let port = portpicker::pick_unused_port().unwrap();
66        // Spawn anvil
67        let provider = ProviderBuilder::new().on_anvil_with_wallet_and_config(|anvil| {
68            anvil.port(port).arg("--accounts").arg("20")
69        })?;
70        let rpc_url = format!("http://localhost:{port}").parse()?;
71        let deployer_address = provider.default_signer_address();
72        // I don't know how to get the signer out of the provider, by default anvil uses the dev
73        // mnemonic and the default signer is the first account.
74        let signer = build_signer(DEV_MNEMONIC.to_string(), 0);
75        assert_eq!(
76            signer.address(),
77            deployer_address,
78            "Signer address mismatch"
79        );
80
81        // Create a fake stake table to create a genesis state. This is fine because we don't
82        // currently use the light client contract. Will need to be updated once we implement
83        // slashing and call the light client contract from the stake table contract.
84        let blocks_per_epoch = 100;
85        let epoch_start_block = 1;
86        let (genesis_state, genesis_stake) = light_client_genesis_from_stake_table(
87            &Default::default(),
88            STAKE_TABLE_CAPACITY_FOR_TEST,
89        )
90        .unwrap();
91
92        let mut contracts = Contracts::new();
93        let args = DeployerArgsBuilder::default()
94            .deployer(provider.clone())
95            .mock_light_client(true)
96            .genesis_lc_state(genesis_state)
97            .genesis_st_state(genesis_stake)
98            .blocks_per_epoch(blocks_per_epoch)
99            .epoch_start_block(epoch_start_block)
100            .multisig_pauser(deployer_address)
101            .exit_escrow_period(U256::from(exit_escrow_period.as_secs()))
102            .token_name("Espresso".to_string())
103            .token_symbol("ESP".to_string())
104            .initial_token_supply(U256::from(3590000000u64))
105            .ops_timelock_delay(U256::from(0))
106            .ops_timelock_admin(signer.address())
107            .ops_timelock_proposers(vec![signer.address()])
108            .ops_timelock_executors(vec![signer.address()])
109            .safe_exit_timelock_delay(U256::from(10))
110            .safe_exit_timelock_admin(signer.address())
111            .safe_exit_timelock_proposers(vec![signer.address()])
112            .safe_exit_timelock_executors(vec![signer.address()])
113            .use_timelock_owner(false)
114            .build()
115            .unwrap();
116
117        match stake_table_contract_version {
118            StakeTableContractVersion::V1 => args.deploy_to_stake_table_v1(&mut contracts).await?,
119            StakeTableContractVersion::V2 => args.deploy_all(&mut contracts).await?,
120        };
121
122        let stake_table = contracts
123            .address(Contract::StakeTableProxy)
124            .expect("StakeTableProxy deployed");
125        let token = contracts
126            .address(Contract::EspTokenProxy)
127            .expect("EspTokenProxy deployed");
128
129        let approval_amount = parse_ether("1000000")?;
130        // Approve the stake table contract so it can transfer tokens to itself
131        let receipt = EspTokenInstance::new(token, &provider)
132            .approve(stake_table, approval_amount)
133            .send()
134            .await?
135            .get_receipt()
136            .await?;
137        assert!(receipt.status());
138
139        let mut rng = StdRng::from_seed([42u8; 32]);
140        let (_, bls_key_pair, state_key_pair) = Self::gen_keys(&mut rng);
141
142        Ok(Self {
143            provider,
144            signer,
145            deployer_address,
146            token,
147            stake_table,
148            exit_escrow_period,
149            rpc_url,
150            bls_key_pair,
151            state_key_pair,
152            commission: Commission::try_from("12.34")?,
153            approval_amount,
154        })
155    }

270    /// Inject test system config into CLI command via arguments
271    pub fn args(&self, cmd: &mut Command, signer: Signer) {
272        cmd.arg("--rpc-url")
273            .arg(self.rpc_url.to_string())
274            .arg("--stake-table-address")
275            .arg(self.stake_table.to_string())
276            .arg("--account-index")
277            .arg("0");
278
279        match signer {
280            Signer::Ledger => cmd.arg("--ledger"),
281            Signer::Mnemonic => cmd.arg("--mnemonic").arg(DEV_MNEMONIC),
282            Signer::BrokeMnemonic => cmd
283                .arg("--mnemonic")
284                .arg("roast term reopen pave choose high rally trouble upon govern hollow stand"),
285        };
286    }

300    #[tokio::test]
301    async fn test_deploy() -> Result<()> {
302        let system = TestSystem::deploy().await?;
303        let stake_table = StakeTable::new(system.stake_table, &system.provider);
304        // sanity check that we can fetch the exit escrow period
305        assert_eq!(
306            stake_table.exitEscrowPeriod().call().await?._0,
307            U256::from(system.exit_escrow_period.as_secs())
308        );
309
310        let to = "0x1111111111111111111111111111111111111111".parse()?;
311
312        // sanity check that we can transfer tokens
313        system.transfer(to, U256::from(123)).await?;
314
315        // sanity check that we can fetch the balance
316        assert_eq!(system.balance(to).await?, U256::from(123));
317
318        Ok(())
319    }
