23/// Returns both genesis light client state and stake table state
24pub async fn light_client_genesis(
25    sequencer_url: &Url,
26    stake_table_capacity: usize,
27) -> anyhow::Result<(LightClientStateSol, StakeTableStateSol)> {
28    let st = fetch_stake_table_from_sequencer(sequencer_url, None)
29        .await
30        .with_context(|| "Failed to initialize stake table")?;
31    light_client_genesis_from_stake_table(&st, stake_table_capacity)
32}

34/// Fetch the stake table from a sequencer node given the epoch number
35///
36/// Does not error, runs until the stake table is provided.
37pub async fn fetch_stake_table_from_sequencer(
38    sequencer_url: &Url,
39    epoch: Option<<SeqTypes as NodeType>::Epoch>,
40) -> Result<HSStakeTable<SeqTypes>> {
41    tracing::info!("Initializing stake table from node for epoch {epoch:?}");
42
43    loop {
44        match epoch {
45            Some(epoch) => match surf_disco::Client::<
46                tide_disco::error::ServerError,
47                StaticVersion<0, 1>,
48            >::new(sequencer_url.clone())
49            .get::<Vec<PeerConfig<SeqTypes>>>(&format!("node/stake-table/{}", epoch.u64()))
50            .send()
51            .await
52            {
53                Ok(resp) => break Ok(resp.into()),
54                Err(e) => {
55                    let url = sequencer_url
56                        .join(&format!("node/stake-table/{}", epoch.u64()))
57                        .unwrap();
58                    tracing::error!(%url, "Failed to fetch the stake table: {e}");
59                    sleep(Duration::from_secs(5)).await;
60                },
61            },
62            None => {
63                let url = sequencer_url.join("config/hotshot").unwrap();
64                match reqwest::get(url.clone()).await {
65                    Ok(resp) => {
66                        let value: serde_json::Value = resp.json().await.with_context(|| {
67                            format!("Failed to parse the json object from url {url}")
68                        })?;
69                        let known_nodes_with_stake =
70                            serde_json::from_str::<Vec<PeerConfig<SeqTypes>>>(
71                                &value["config"]["known_nodes_with_stake"].to_string(),
72                            )
73                            .with_context(|| "Failed to parse the stake table")?;
74                        break Ok(known_nodes_with_stake.into());
75                    },
76                    Err(e) => {
77                        tracing::error!(%url, "Failed to fetch the network config: {e}");
78                        sleep(Duration::from_secs(5)).await;
79                    },
80                }
81            },
82        }
83    }
84}

87/// derive the genesis light client state and stake table state from initial set of `PeerConfig`
88pub fn light_client_genesis_from_stake_table(
89    st: &HSStakeTable<SeqTypes>,
90    stake_table_capacity: usize,
91) -> anyhow::Result<(LightClientStateSol, StakeTableStateSol)> {
92    let st_state = st.commitment(stake_table_capacity)?;
93    Ok((
94        LightClientStateSol {
95            viewNum: 0,
96            blockHeight: 0,
97            blockCommRoot: U256::from(0u32),
98        },
99        StakeTableStateSol {
100            blsKeyComm: field_to_u256(st_state.bls_key_comm),
101            schnorrKeyComm: field_to_u256(st_state.schnorr_key_comm),
102            amountComm: field_to_u256(st_state.amount_comm),
103            threshold: field_to_u256(st_state.threshold),
104        },
105    ))
106}

108/// Get the epoch-related  from the sequencer's `PublicHotShotConfig` struct
109/// return (blocks_per_epoch, epoch_start_block)
110pub async fn fetch_epoch_config_from_sequencer(sequencer_url: &Url) -> anyhow::Result<(u64, u64)> {
111    // Request the configuration until it is successful
112    loop {
113        let url = sequencer_url.join("config/hotshot").unwrap();
114        match reqwest::get(url.clone()).await {
115            Ok(resp) => {
116                let value: serde_json::Value = resp
117                    .json()
118                    .await
119                    .with_context(|| format!("Failed to parse the json object from url {url}"))?;
120                let blocks_per_epoch =
121                    value["config"]["epoch_height"]
122                        .as_u64()
123                        .ok_or(anyhow::anyhow!(
124                            "Failed to parse epoch_height from hotshot config"
125                        ))?;
126                let epoch_start_block =
127                    value["config"]["epoch_start_block"]
128                        .as_u64()
129                        .ok_or(anyhow::anyhow!(
130                            "Failed to parse epoch_start_block from hotshot config"
131                        ))?;
132                break Ok((blocks_per_epoch, epoch_start_block));
133            },
134            Err(e) => {
135                tracing::error!(%url, "Failed to fetch the network config: {e}");
136                sleep(Duration::from_secs(5)).await;
137            },
138        }
139    }
