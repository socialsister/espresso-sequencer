28/// Convenient handler that builds all the input arguments ready to be deployed.
29/// - `deployer`: deployer's wallet provider
30/// - `token_recipient`: initial token holder, same as deployer if None.
31/// - `mock_light_client`: flag to indicate whether deploying mocked contract
32/// - `genesis_lc_state`: Genesis light client state
33/// - `genesis_st_state`: Genesis stake table state
34/// - `permissioned_prover`: permissioned light client prover address
35/// - `blocks_per_epoch`: epoch length in block height
36/// - `epoch_start_block`: block height for the first *activated* epoch
37/// - `exit_escrow_period`: exit escrow period for stake table (in seconds)
38/// - `multisig`: new owner/multisig that owns all the proxy contracts
39/// - `multisig_pauser`: new multisig that owns the pauser role
40/// - `initial_token_supply`: initial token supply for the token contract
41/// - `token_name`: name of the token
42/// - `token_symbol`: symbol of the token
43/// - `ops_timelock_admin`: admin address for the ops timelock
44/// - `ops_timelock_delay`: delay for the ops timelock
45/// - `ops_timelock_executors`: executors for the ops timelock
46/// - `ops_timelock_proposers`: proposers for the ops timelock
47/// - `safe_exit_timelock_admin`: admin address for the safe exit timelock
48/// - `safe_exit_timelock_delay`: delay for the safe exit timelock
49/// - `safe_exit_timelock_executors`: executors for the safe exit timelock
50/// - `safe_exit_timelock_proposers`: proposers for the safe exit timelock
51/// - `timelock_operation_type`: type of the timelock operation
52/// - `timelock_target_contract`: target contract for the timelock operation
53/// - `timelock_operation_value`: value for the timelock operation
54/// - `timelock_operation_delay`: delay for the timelock operation
55/// - `timelock_operation_function_signature`: function signature for the timelock operation
56/// - `timelock_operation_function_values`: function values for the timelock operation
57/// - `timelock_operation_salt`: salt for the timelock operation
58/// - `use_timelock_owner`: flag to indicate whether to transfer ownership to the timelock owner
59#[derive(Builder, Clone)]
60#[builder(setter(strip_option))]

130    /// deploy target contracts
131    pub async fn deploy(&self, contracts: &mut Contracts, target: Contract) -> Result<()> {
132        let provider = &self.deployer;
133        let admin = provider.default_signer_address();
134        match target {
135            Contract::FeeContractProxy => {
136                let addr = crate::deploy_fee_contract_proxy(provider, contracts, admin).await?;
137
138                if let Some(use_timelock_owner) = self.use_timelock_owner {
139                    // FeeContract uses OpsTimelock because:
140                    // - It handles critical fee collection and distribution logic
141                    // - May require emergency updates for security or functionality
142                    // - OpsTimelock provides a shorter delay for critical operations
143                    tracing::info!(
144                        "Transferring ownership to OpsTimelock: {:?}",
145                        use_timelock_owner
146                    );
147                    // deployer is the timelock owner
148                    if use_timelock_owner {
149                        let timelock_addr = contracts
150                            .address(Contract::OpsTimelock)
151                            .expect("fail to get OpsTimelock address");
152                        crate::transfer_ownership(provider, target, addr, timelock_addr).await?;
153                    }
154                } else if let Some(multisig) = self.multisig {
155                    tracing::info!("Transferring ownership to multisig: {:?}", multisig);
156                    crate::transfer_ownership(provider, target, addr, multisig).await?;
157                }
158            },
159            Contract::EspTokenProxy => {
160                let token_recipient = self.token_recipient.unwrap_or(admin);
161                let token_name = self
162                    .token_name
163                    .clone()
164                    .context("Token name must be set when deploying esp token")?;
165                let token_symbol = self
166                    .token_symbol
167                    .clone()
168                    .context("Token symbol must be set when deploying esp token")?;
169                let initial_supply = self
170                    .initial_token_supply
171                    .context("Initial token supply must be set when deploying esp token")?;
172                let addr = crate::deploy_token_proxy(
173                    provider,
174                    contracts,
175                    admin,
176                    token_recipient,
177                    initial_supply,
178                    &token_name,
179                    &token_symbol,
180                )
181                .await?;
182
183                if let Some(use_timelock_owner) = self.use_timelock_owner {
184                    // EspToken uses SafeExitTimelock (not OpsTimelock) because:
185                    // - It's a simple ERC20 token with minimal upgrade complexity
186                    // - No emergency updates are expected for token functionality
187                    // - SafeExitTimelock provides sufficient security for token operations
188                    tracing::info!("Transferring ownership to SafeExitTimelock");
189                    // deployer is the timelock owner
190                    if use_timelock_owner {
191                        let timelock_addr = contracts
192                            .address(Contract::SafeExitTimelock)
193                            .expect("fail to get SafeExitTimelock address");
194                        crate::transfer_ownership(provider, target, addr, timelock_addr).await?;
195                    }
196                } else if let Some(multisig) = self.multisig {
197                    crate::transfer_ownership(provider, target, addr, multisig).await?;
198                }
199            },
200            Contract::EspTokenV2 => {
201                let use_multisig = self.use_multisig;
202
203                if use_multisig {
204                    upgrade_esp_token_v2_multisig_owner(
205                        provider,
206                        contracts,
207                        self.rpc_url.clone(),
208                        Some(self.dry_run),
209                    )
210                    .await?;
211                } else {
212                    crate::upgrade_esp_token_v2(provider, contracts).await?;
213                    let addr = contracts
214                        .address(Contract::EspTokenProxy)
215                        .expect("fail to get EspTokenProxy address");
216
217                    if let Some(use_timelock_owner) = self.use_timelock_owner {
218                        tracing::info!("Transferring ownership to SafeExitTimelock");
219                        // deployer is the timelock owner
220                        if use_timelock_owner {
221                            let timelock_addr = contracts
222                                .address(Contract::SafeExitTimelock)
223                                .expect("fail to get SafeExitTimelock address");
224                            crate::transfer_ownership(provider, target, addr, timelock_addr)
225                                .await?;
226                        }
227                    } else if let Some(multisig) = self.multisig {
228                        let token_proxy = contracts
229                            .address(Contract::EspTokenProxy)
230                            .expect("fail to get EspTokenProxy address");
231                        crate::transfer_ownership(
232                            provider,
233                            Contract::EspTokenProxy,
234                            token_proxy,
235                            multisig,
236                        )
237                        .await?;
238                    }
239                }
240            },
241            Contract::LightClientProxy => {
242                assert!(
243                    self.genesis_lc_state.is_some(),
244                    "forget to specify genesis_lc_state()"
245                );
246                assert!(
247                    self.genesis_st_state.is_some(),
248                    "forget to specify genesis_st_state()"
249                );
250                crate::deploy_light_client_proxy(
251                    provider,
252                    contracts,
253                    self.mock_light_client,
254                    self.genesis_lc_state.clone().unwrap(),
255                    self.genesis_st_state.clone().unwrap(),
256                    admin,
257                    self.permissioned_prover,
258                )
259                .await?;
260                // NOTE: we don't transfer ownership to multisig, we only do so after V2 upgrade
261            },
262            Contract::LightClientV2 => {
263                assert!(
264                    self.blocks_per_epoch.is_some(),
265                    "forget to specify blocks_per_epoch()"
266                );
267                assert!(
268                    self.epoch_start_block.is_some(),
269                    "forget to specify epoch_start_block()"
270                );
271
272                let use_mock = self.mock_light_client;
273                let dry_run = self.dry_run;
274                let use_multisig = self.use_multisig;
275                let mut blocks_per_epoch = self.blocks_per_epoch.unwrap();
276                let epoch_start_block = self.epoch_start_block.unwrap();
277                let rpc_url = self.rpc_url.clone();
278
279                // TEST-ONLY: if this config is not yet set, we use a large default value
280                // to avoid contract complaining about invalid zero-valued blocks_per_epoch.
281                // This large value will act as if we are always in epoch 1, which won't conflict
282                // with the effective purpose of the real `PublicNetworkConfig`.
283                if use_mock && blocks_per_epoch == 0 {
284                    blocks_per_epoch = u64::MAX;
285                }
286                tracing::info!(%blocks_per_epoch, ?dry_run, ?use_multisig, "Upgrading LightClientV2 with ");
287                if use_multisig {
288                    upgrade_light_client_v2_multisig_owner(
289                        provider,
290                        contracts,
291                        LightClientV2UpgradeParams {
292                            blocks_per_epoch,
293                            epoch_start_block,
294                        },
295                        use_mock,
296                        rpc_url,
297                        Some(dry_run),
298                    )
299                    .await?;
300                } else {
301                    crate::upgrade_light_client_v2(
302                        provider,
303                        contracts,
304                        use_mock,
305                        blocks_per_epoch,
306                        epoch_start_block,
307                    )
308                    .await?;
309
310                    let addr = contracts
311                        .address(Contract::LightClientProxy)
312                        .expect("fail to get LightClientProxy address");
313
314                    if let Some(use_timelock_owner) = self.use_timelock_owner {
315                        // LightClient uses OpsTimelock because:
316                        // - It's a critical security component for the network
317                        // - May require emergency updates for security vulnerabilities
318                        // - OpsTimelock provides a shorter delay for critical operations
319                        tracing::info!("Transferring ownership to OpsTimelock");
320                        // deployer is the timelock owner
321                        if use_timelock_owner {
322                            let timelock_addr = contracts
323                                .address(Contract::OpsTimelock)
324                                .expect("fail to get OpsTimelock address");
325                            crate::transfer_ownership(provider, target, addr, timelock_addr)
326                                .await?;
327                        }
328                    } else if let Some(multisig) = self.multisig {
329                        let lc_proxy = contracts
330                            .address(Contract::LightClientProxy)
331                            .expect("fail to get LightClientProxy address");
332                        crate::transfer_ownership(
333                            provider,
334                            Contract::LightClientProxy,
335                            lc_proxy,
336                            multisig,
337                        )
338                        .await?;
339                    }
340                }
341            },
342            Contract::StakeTableProxy => {
343                let token_addr = contracts
344                    .address(Contract::EspTokenProxy)
345                    .context("no ESP token proxy address")?;
346                let lc_addr = contracts
347                    .address(Contract::LightClientProxy)
348                    .context("no LightClient proxy address")?;
349                let escrow_period = self.exit_escrow_period.unwrap_or(U256::from(300));
350                crate::deploy_stake_table_proxy(
351                    provider,
352                    contracts,
353                    token_addr,
354                    lc_addr,
355                    escrow_period,
356                    admin,
357                )
358                .await?;
359
360                // NOTE: we don't transfer ownership to multisig, we only do so after V2 upgrade
361            },
362            Contract::StakeTableV2 => {
363                let use_multisig = self.use_multisig;
364                let dry_run = self.dry_run;
365                let multisig_pauser = self.multisig_pauser.context(
366                    "Multisig pauser address must be set for the upgrade to StakeTableV2",
367                )?;
368                tracing::info!(?dry_run, ?use_multisig, "Upgrading to StakeTableV2 with ");
369                if use_multisig {
370                    upgrade_stake_table_v2_multisig_owner(
371                        provider,
372                        contracts,
373                        self.rpc_url.clone(),
374                        self.multisig.context(
375                            "Multisig address must be set when upgrading to --use-multisig flag \
376                             is present",
377                        )?,
378                        multisig_pauser,
379                        Some(dry_run),
380                    )
381                    .await?;
382                } else {
383                    crate::upgrade_stake_table_v2(provider, contracts, multisig_pauser, admin)
384                        .await?;
385
386                    let addr = contracts
387                        .address(Contract::StakeTableProxy)
388                        .expect("fail to get StakeTableProxy address");
389
390                    if let Some(use_timelock_owner) = self.use_timelock_owner {
391                        // StakeTable uses OpsTimelock because:
392                        // - It manages critical staking and validator operations
393                        // - May require emergency updates for security or functionality
394                        // - OpsTimelock provides a shorter delay for critical operations
395                        tracing::info!("Transferring ownership to OpsTimelock");
396                        // deployer is the timelock owner
397                        if use_timelock_owner {
398                            let timelock_addr = contracts
399                                .address(Contract::OpsTimelock)
400                                .expect("fail to get OpsTimelock address");
401                            crate::transfer_ownership(provider, target, addr, timelock_addr)
402                                .await?;
403                        }
404                    } else if let Some(multisig) = self.multisig {
405                        let stake_table_proxy = contracts
406                            .address(Contract::StakeTableProxy)
407                            .expect("fail to get StakeTableProxy address");
408                        crate::transfer_ownership(
409                            provider,
410                            Contract::StakeTableProxy,
411                            stake_table_proxy,
412                            multisig,
413                        )
414                        .await?;
415                    }
416                }
417            },
418            Contract::OpsTimelock => {
419                let ops_timelock_delay = self
420                    .ops_timelock_delay
421                    .context("Ops Timelock delay must be set when deploying Ops Timelock")?;
422                let ops_timelock_proposers = self
423                    .ops_timelock_proposers
424                    .clone()
425                    .context("Ops Timelock proposers must be set when deploying Ops Timelock")?;
426                let ops_timelock_executors = self
427                    .ops_timelock_executors
428                    .clone()
429                    .context("Ops Timelock executors must be set when deploying Ops Timelock")?;
430                let ops_timelock_admin = self
431                    .ops_timelock_admin
432                    .context("Ops Timelock admin must be set when deploying Ops Timelock")?;
433                crate::deploy_ops_timelock(
434                    provider,
435                    contracts,
436                    ops_timelock_delay,
437                    ops_timelock_proposers,
438                    ops_timelock_executors,
439                    ops_timelock_admin,
440                )
441                .await?;
442            },
443            Contract::SafeExitTimelock => {
444                let safe_exit_timelock_delay = self.safe_exit_timelock_delay.context(
445                    "SafeExitTimelock delay must be set when deploying SafeExitTimelock",
446                )?;
447                let safe_exit_timelock_proposers =
448                    self.safe_exit_timelock_proposers.clone().context(
449                        "SafeExitTimelock proposers must be set when deploying SafeExitTimelock",
450                    )?;
451                let safe_exit_timelock_executors =
452                    self.safe_exit_timelock_executors.clone().context(
453                        "SafeExitTimelock executors must be set when deploying SafeExitTimelock",
454                    )?;
455                let safe_exit_timelock_admin = self.safe_exit_timelock_admin.context(
456                    "SafeExitTimelock admin must be set when deploying SafeExitTimelock",
457                )?;
458                crate::deploy_safe_exit_timelock(
459                    provider,
460                    contracts,
461                    safe_exit_timelock_delay,
462                    safe_exit_timelock_proposers,
463                    safe_exit_timelock_executors,
464                    safe_exit_timelock_admin,
465                )
466                .await?;
467            },
468            _ => {
469                panic!("Deploying {target} not supported.");
470            },
471        }
472        Ok(())
473    }

475    /// Deploy all contracts up to and including stake table v1
476    pub async fn deploy_to_stake_table_v1(&self, contracts: &mut Contracts) -> Result<()> {
477        // Deploy timelocks first so they can be used as owners for other contracts
478        self.deploy(contracts, Contract::OpsTimelock).await?;
479        self.deploy(contracts, Contract::SafeExitTimelock).await?;
480
481        // Then deploy other contracts
482        self.deploy(contracts, Contract::FeeContractProxy).await?;
483        self.deploy(contracts, Contract::EspTokenProxy).await?;
484        self.deploy(contracts, Contract::LightClientProxy).await?;
485        self.deploy(contracts, Contract::LightClientV2).await?;
486        self.deploy(contracts, Contract::StakeTableProxy).await?;
487        Ok(())
488    }

493        self.deploy(contracts, Contract::StakeTableV2).await?;
494        Ok(())

497    // Perform a timelock operation
498    ///
499    /// Parameters:
500    /// - `contracts`: ref to deployed contracts
501    ///
502    pub async fn perform_timelock_operation_on_contract(
503        &self,
504        contracts: &mut Contracts,
505    ) -> Result<()> {
506        let timelock_operation_type = self
507            .timelock_operation_type
508            .context("Timelock operation type not found")?;
509        let target_contract = self
510            .timelock_target_contract
511            .clone()
512            .context("Timelock target not found")?;
513        let value = self
514            .timelock_operation_value
515            .context("Timelock operation value not found")?;
516        let function_signature = self
517            .timelock_operation_function_signature
518            .as_ref()
519            .context("Timelock operation function signature not found")?;
520        let function_values = self
521            .timelock_operation_function_values
522            .clone()
523            .context("Timelock operation function values not found")?;
524        let salt = self
525            .timelock_operation_salt
526            .clone()
527            .context("Timelock operation salt not found")?;
528        let delay = self
529            .timelock_operation_delay
530            .context("Timelock operation delay not found")?;
531
532        let (target_addr, contract_type) = match target_contract.as_str() {
533            "FeeContract" => (
534                contracts
535                    .address(Contract::FeeContractProxy)
536                    .context("FeeContractProxy address not found")?,
537                Contract::FeeContractProxy,
538            ),
539            "EspToken" => (
540                contracts
541                    .address(Contract::EspTokenProxy)
542                    .context("EspTokenProxy address not found")?,
543                Contract::EspTokenProxy,
544            ),
545            "LightClient" => (
546                contracts
547                    .address(Contract::LightClientProxy)
548                    .context("LightClientProxy address not found")?,
549                Contract::LightClientProxy,
550            ),
551            "StakeTable" => (
552                contracts
553                    .address(Contract::StakeTableProxy)
554                    .context("StakeTableProxy address not found")?,
555                Contract::StakeTableProxy,
556            ),
557            _ => anyhow::bail!("Invalid target contract: {}", target_contract),
558        };
559
560        let function_calldata = encode_function_call(function_signature, function_values.clone())
561            .context("Failed to encode function data")?;
562
563        // Parse salt from string to B256
564        let salt_bytes = if salt == "0x" || salt.is_empty() {
565            B256::ZERO // Use zero salt if empty
566        } else if let Some(stripped) = salt.strip_prefix("0x") {
567            B256::from_hex(stripped).context("Invalid salt hex format")?
568        } else {
569            B256::from_hex(&salt).context("Invalid salt hex format")?
570        };
571
572        let timelock_operation_data = TimelockOperationData {
573            target: target_addr,
574            value,
575            data: function_calldata,
576            predecessor: B256::ZERO, // Default to no predecessor
577            salt: salt_bytes,
578            delay,
579        };
580
581        match timelock_operation_type {
582            TimelockOperationType::Schedule => {
583                let operation_id = schedule_timelock_operation(
584                    &self.deployer,
585                    contract_type,
586                    timelock_operation_data,
587                )
588                .await?;
589                tracing::info!("Timelock operation scheduled with ID: {}", operation_id);
590            },
591            TimelockOperationType::Execute => {
592                let tx_id = execute_timelock_operation(
593                    &self.deployer,
594                    contract_type,
595                    timelock_operation_data,
596                )
597                .await?;
598                tracing::info!("Timelock operation executed with ID: {}", tx_id);
599            },
600            TimelockOperationType::Cancel => {
601                let tx_id = cancel_timelock_operation(
602                    &self.deployer,
603                    contract_type,
604                    timelock_operation_data,
605                )
606                .await?;
607                tracing::info!("Timelock operation cancelled with ID: {}", tx_id);
608            },
609        }
610        Ok(())
611    }

613    /// Propose ownership transfer from multisig to timelock
614    pub async fn propose_transfer_ownership_from_multisig_to_timelock(
615        &self,
616        contracts: &mut Contracts,
617        timelock_controller: Address,
618        contract: Contract,
619    ) -> Result<()> {
620        let multisig = self.multisig.expect(
621            "Multisig address must be set when proposing ownership transfer. Use \
622             --multisig-address or ESPRESSO_SEQUENCER_ETH_MULTISIG_ADDRESS",
623        );
624        let rpc_url = self.rpc_url.clone();
625        let dry_run = self.dry_run;
626        let use_hardware_wallet = false;
627        let result = transfer_ownership_from_multisig_to_timelock(
628            &self.deployer,
629            contracts,
630            contract,
631            TransferOwnershipParams {
632                new_owner: timelock_controller,
633                rpc_url,
634                safe_addr: multisig,
635                use_hardware_wallet,
636                dry_run,
637            },
638        )
639        .await?;
640        if !result.status.success() {
641            let stderr = String::from_utf8_lossy(&result.stderr);
642            let stdout = String::from_utf8_lossy(&result.stdout);
643            anyhow::bail!(
644                "Failed to propose ownership transfer for {}: {}\nStdout: {}\nStderr: {}",
645                contract,
646                result.status,
647                stdout,
648                stderr
649            );
650        }
651        tracing::info!("Successfully proposed ownership transfer for {}", contract);
652        Ok(())
653    }
