27/// Deploy contracts needed to run the sequencer.
28///
29/// This script deploys contracts needed to run the sequencer to an L1. It outputs a .env file
30/// containing the addresses of the deployed contracts.
31///
32/// This script can also be used to do incremental deployments. The only contract addresses
33/// needed to configure the sequencer network are ESPRESSO_SEQUENCER_FEE_CONTRACT_PROXY_ADDRESS,
34/// ESPRESSO_SEQUENCER_LIGHT_CLIENT_PROXY_ADDRESS and ESPRESSO_SEQUENCER_STAKE_TABLE_ADDRESS.
35/// These contracts, however, have dependencies, and a full deployment involves several
36/// contracts. Some of these contracts, especially libraries may already have been deployed, or
37/// perhaps one of the top-level contracts has been deployed and we only need to deploy the other
38/// one.
39///
40/// It is possible to pass in the addresses of already deployed contracts, in which case those
41/// addresses will be used in place of deploying a new contract wherever that contract is required
42/// in the deployment process. The generated .env file will include all the addresses passed in as
43/// well as those newly deployed.

350enum Command {

356#[tokio::main]
357async fn main() -> anyhow::Result<()> {
358    let opt = Options::parse();
359
360    opt.logging.init();
361
362    if matches!(opt.command, Some(Command::VerifyNodeJsFiles)) {
363        verify_node_js_files().await?;
364        return Ok(());
365    };
366
367    let mut contracts = Contracts::from(opt.contracts);
368    let provider = if opt.ledger {
369        let signer = connect_ledger(opt.account_index as usize).await?;
370        tracing::info!("Using ledger for signing, watch ledger device for prompts.");
371        build_provider_ledger(signer, opt.rpc_url.clone(), Some(opt.l1_polling_interval))
372    } else {
373        build_provider(
374            opt.mnemonic
375                .expect("Mnemonic provided when not using ledger"),
376            opt.account_index,
377            opt.rpc_url.clone(),
378            Some(opt.l1_polling_interval),
379        )
380    };
381
382    // Fail early if we can't connect to the Ethereum RPC.
383    let chain_id = provider.get_chain_id().await.with_context(|| {
384        // The URL may contain a key in the query string.
385        let mut url = opt.rpc_url.clone();
386        url.set_query(None);
387        format!("Unable query Ethereum provider {url}..")
388    })?;
389    tracing::info!("Connected to chain with chain ID: {chain_id}");
390
391    let account = provider.default_signer_address();
392    if let Some(command) = &opt.command {
393        match command {
394            Command::Account => {
395                println!("{account}");
396                return Ok(());
397            },
398            Command::Balance => {
399                let balance = provider.get_balance(account).await?;
400                println!("{account}: {} Eth", format_ether(balance));
401                return Ok(());
402            },
403            _ => unreachable!(),
404        };
405    };
406
407    // No subcommand specified. Deploy contracts.
408
409    let balance = provider.get_balance(account).await?;
410    tracing::info!(
411        "Using deployer account {account} with balance: {}",
412        format_ether(balance),
413    );
414    if balance.is_zero() {
415        anyhow::bail!(
416            "account_index {}, address={account} has no balance. A funded account is required.",
417            opt.account_index
418        );
419    }
420
421    // First use builder to build constructor input arguments
422    let mut args_builder = DeployerArgsBuilder::default();
423    args_builder
424        .deployer(provider)
425        .mock_light_client(opt.use_mock)
426        .use_multisig(opt.use_multisig)
427        .dry_run(opt.dry_run)
428        .rpc_url(opt.rpc_url.to_string());
429    if let Some(multisig) = opt.multisig_address {
430        args_builder.multisig(multisig);
431    }
432    if let Some(multisig_pauser) = opt.multisig_pauser_address {
433        args_builder.multisig_pauser(multisig_pauser);
434    }
435
436    if let Some(blocks_per_epoch) = opt.blocks_per_epoch {
437        args_builder.blocks_per_epoch(blocks_per_epoch);
438    }
439    if let Some(epoch_start_block) = opt.epoch_start_block {
440        args_builder.epoch_start_block(epoch_start_block);
441    }
442
443    if opt.deploy_light_client_v1 {
444        let (genesis_state, genesis_stake) = if opt.mock_espresso_live_network {
445            light_client_genesis_from_stake_table(&Default::default(), DEFAULT_STAKE_TABLE_CAPACITY)
446                .unwrap()
447        } else {
448            light_client_genesis(&opt.sequencer_url, opt.stake_table_capacity).await?
449        };
450        args_builder
451            .genesis_lc_state(genesis_state)
452            .genesis_st_state(genesis_stake);
453        if let Some(prover) = opt.permissioned_prover {
454            args_builder.permissioned_prover(prover);
455        }
456    }
457    if opt.upgrade_light_client_v2 {
458        let (blocks_per_epoch, epoch_start_block) =
459            if (opt.dry_run && opt.use_multisig) || opt.mock_espresso_live_network {
460                (10, 22)
461            } else {
462                // fetch epoch length from HotShot config
463                // Request the configuration until it is successful
464                loop {
465                    match surf_disco::Client::<ServerError, StaticVersion<0, 1>>::new(
466                        opt.sequencer_url.clone(),
467                    )
468                    .get::<PublicNetworkConfig>("config/hotshot")
469                    .send()
470                    .await
471                    {
472                        Ok(resp) => {
473                            let config = resp.hotshot_config();
474                            break (config.blocks_per_epoch(), config.epoch_start_block());
475                        },
476                        Err(e) => {
477                            tracing::error!("Failed to fetch the network config: {e}");
478                            sleep(Duration::from_secs(5));
479                        },
480                    }
481                }
482            };
483        args_builder.blocks_per_epoch(blocks_per_epoch);
484        args_builder.epoch_start_block(epoch_start_block);
485    }
486
487    if opt.deploy_stake_table {
488        if let Some(escrow_period) = opt.exit_escrow_period {
489            args_builder.exit_escrow_period(U256::from(escrow_period.as_secs()));
490        }
491    }
492    if opt.deploy_ops_timelock {
493        let ops_timelock_admin = opt.ops_timelock_admin.ok_or_else(|| {
494            anyhow::anyhow!(
495                "Must provide --ops-timelock-admin or ESPRESSO_OPS_TIMELOCK_ADMIN env var when \
496                 deploying ops timelock"
497            )
498        })?;
499        args_builder.ops_timelock_admin(ops_timelock_admin);
500        let ops_timelock_delay = opt.ops_timelock_delay.ok_or_else(|| {
501            anyhow::anyhow!(
502                "Must provide --ops-timelock-delay or ESPRESSO_OPS_TIMELOCK_DELAY env var when \
503                 deploying ops timelock"
504            )
505        })?;
506        args_builder.ops_timelock_delay(U256::from(ops_timelock_delay));
507        let ops_timelock_executors = opt.ops_timelock_executors.ok_or_else(|| {
508            anyhow::anyhow!(
509                "Must provide --ops-timelock-executors or ESPRESSO_OPS_TIMELOCK_EXECUTORS env var \
510                 when deploying ops timelock"
511            )
512        })?;
513        args_builder.ops_timelock_executors(ops_timelock_executors.into_iter().collect());
514        let ops_timelock_proposers = opt.ops_timelock_proposers.ok_or_else(|| {
515            anyhow::anyhow!(
516                "Must provide --ops-timelock-proposers or ESPRESSO_OPS_TIMELOCK_PROPOSERS env var \
517                 when deploying ops timelock"
518            )
519        })?;
520        args_builder.ops_timelock_proposers(ops_timelock_proposers.into_iter().collect());
521    }
522
523    if opt.deploy_safe_exit_timelock {
524        let safe_exit_timelock_admin = opt.safe_exit_timelock_admin.ok_or_else(|| {
525            anyhow::anyhow!(
526                "Must provide --safe-exit-timelock-admin or ESPRESSO_SAFE_EXIT_TIMELOCK_ADMIN env \
527                 var when deploying safe exit timelock"
528            )
529        })?;
530        args_builder.safe_exit_timelock_admin(safe_exit_timelock_admin);
531        let safe_exit_timelock_delay = opt.safe_exit_timelock_delay.ok_or_else(|| {
532            anyhow::anyhow!(
533                "Must provide --safe-exit-timelock-delay or ESPRESSO_SAFE_EXIT_TIMELOCK_DELAY env \
534                 var when deploying safe exit timelock"
535            )
536        })?;
537        args_builder.safe_exit_timelock_delay(U256::from(safe_exit_timelock_delay));
538        let safe_exit_timelock_executors = opt.safe_exit_timelock_executors.ok_or_else(|| {
539            anyhow::anyhow!(
540                "Must provide --safe-exit-timelock-executors or \
541                 ESPRESSO_SAFE_EXIT_TIMELOCK_EXECUTORS env var when deploying safe exit timelock"
542            )
543        })?;
544        args_builder
545            .safe_exit_timelock_executors(safe_exit_timelock_executors.into_iter().collect());
546        let safe_exit_timelock_proposers = opt.safe_exit_timelock_proposers.ok_or_else(|| {
547            anyhow::anyhow!(
548                "Must provide --safe-exit-timelock-proposers or \
549                 ESPRESSO_SAFE_EXIT_TIMELOCK_PROPOSERS env var when deploying safe exit timelock"
550            )
551        })?;
552        args_builder
553            .safe_exit_timelock_proposers(safe_exit_timelock_proposers.into_iter().collect());
554    }
555    if opt.use_timelock_owner {
556        args_builder.use_timelock_owner(true);
557    }
558
559    if opt.perform_timelock_operation {
560        let timelock_operation_type = opt.timelock_operation_type.ok_or_else(|| {
561            anyhow::anyhow!(
562                "Must provide --timelock-operation-type or ESPRESSO_TIMELOCK_OPERATION_TYPE env \
563                 var when scheduling timelock operation"
564            )
565        })?;
566        args_builder.timelock_operation_type(timelock_operation_type);
567        let timelock_target_contract = opt.timelock_target_contract.clone().ok_or_else(|| {
568            anyhow::anyhow!(
569                "Must provide --timelock-target-contract or \
570                 ESPRESSO_TIMELOCK_OPERATION_TARGET_CONTRACT env var when scheduling timelock \
571                 operation"
572            )
573        })?;
574        args_builder.timelock_target_contract(timelock_target_contract);
575        let function_signature = opt.function_signature.ok_or_else(|| {
576            anyhow::anyhow!(
577                "Must provide --function-signature or \
578                 ESPRESSO_TIMELOCK_OPERATION_FUNCTION_SIGNATURE env var when performing timelock \
579                 operation"
580            )
581        })?;
582        args_builder.timelock_operation_function_signature(function_signature);
583        let function_values = opt.function_values.ok_or_else(|| {
584            anyhow::anyhow!(
585                "Must provide --function-values or ESPRESSO_TIMELOCK_OPERATION_FUNCTION_VALUES \
586                 env var when performing timelock operation"
587            )
588        })?;
589        args_builder.timelock_operation_function_values(function_values);
590        let timelock_operation_salt = opt.timelock_operation_salt.ok_or_else(|| {
591            anyhow::anyhow!(
592                "Must provide --timelock-operation-salt or ESPRESSO_TIMELOCK_OPERATION_SALT env \
593                 var when scheduling timelock operation"
594            )
595        })?;
596        args_builder.timelock_operation_salt(timelock_operation_salt);
597        let timelock_operation_delay = opt.timelock_operation_delay.ok_or_else(|| {
598            anyhow::anyhow!(
599                "Must provide --timelock-operation-delay or ESPRESSO_TIMELOCK_OPERATION_DELAY env \
600                 var when scheduling timelock operation"
601            )
602        })?;
603        args_builder.timelock_operation_delay(U256::from(timelock_operation_delay));
604        let timelock_operation_value = opt.timelock_operation_value.unwrap_or_default();
605        args_builder.timelock_operation_value(timelock_operation_value);
606    }
607
608    if opt.deploy_esp_token {
609        let token_recipient = opt
610            .initial_token_grant_recipient
611            .expect("Must provide --initial-token-grant-recipient when deploying esp token");
612        let token_name = opt
613            .token_name
614            .expect("Must provide --token-name when deploying esp token");
615        let token_symbol = opt
616            .token_symbol
617            .expect("Must provide --token-symbol when deploying esp token");
618        let initial_token_supply = opt
619            .initial_token_supply
620            .expect("Must provide --initial-token-supply when deploying esp token");
621        args_builder.token_name(token_name);
622        args_builder.token_symbol(token_symbol);
623        args_builder.initial_token_supply(initial_token_supply);
624        args_builder.token_recipient(token_recipient);
625    }
626
627    // then deploy specified contracts
628    let args = args_builder.build()?;
629
630    // Deploy timelocks first so they can be used as owners for other contracts
631    if opt.deploy_ops_timelock {
632        args.deploy(&mut contracts, Contract::OpsTimelock).await?;
633    }
634    if opt.deploy_safe_exit_timelock {
635        args.deploy(&mut contracts, Contract::SafeExitTimelock)
636            .await?;
637    }
638
639    // Then deploy other contracts
640    if opt.deploy_fee {
641        args.deploy(&mut contracts, Contract::FeeContractProxy)
642            .await?;
643    }
644    if opt.deploy_esp_token {
645        args.deploy(&mut contracts, Contract::EspTokenProxy).await?;
646    }
647    if opt.upgrade_esp_token_v2 {
648        args.deploy(&mut contracts, Contract::EspTokenV2).await?;
649    }
650    if opt.deploy_light_client_v1 {
651        args.deploy(&mut contracts, Contract::LightClientProxy)
652            .await?;
653    }
654    if opt.upgrade_light_client_v2 {
655        args.deploy(&mut contracts, Contract::LightClientV2).await?;
656    }
657    if opt.deploy_stake_table {
658        args.deploy(&mut contracts, Contract::StakeTableProxy)
659            .await?;
660    }
661    if opt.upgrade_stake_table_v2 {
662        args.deploy(&mut contracts, Contract::StakeTableV2).await?;
663    }
664
665    // then perform the timelock operation if any
666    if opt.perform_timelock_operation {
667        args.perform_timelock_operation_on_contract(&mut contracts)
668            .await?;
669    }
670
671    // Execute ownership transfer proposal if requested
672    if opt.propose_transfer_ownership_to_timelock_fee_contract {
673        let timelock_controller = opt.timelock_controller.expect(
674            "Must provide --timelock-controller when transferring ownership from multisig to \
675             timelock",
676        );
677
678        args.propose_transfer_ownership_from_multisig_to_timelock(
679            &mut contracts,
680            timelock_controller,
681            Contract::FeeContractProxy,
682        )
683        .await?;
684    }
685
686    // finally print out or persist deployed addresses
687    if let Some(out) = &opt.out {
688        let file = File::options()
689            .create(true)
690            .truncate(true)
691            .write(true)
692            .open(out)?;
693        contracts.write(file)?;
694    } else {
695        contracts.write(stdout())?;
696    }
697
698    Ok(())
699}
