1//! Sequencer node persistence.
2//!
3//! This module implements the persistence required for a sequencer node to rejoin the network and
4//! resume participating in consensus, in the event that its process crashes or is killed and loses
5//! all in-memory state.
6//!
7//! This is distinct from the query service persistent storage found in the `api` module, which is
8//! an extension that node operators can opt into. This module defines the minimum level of
9//! persistence which is _required_ to run a node.

123    impl EventCollector {
124        async fn leaf_chain(&self) -> Vec<LeafInfo<SeqTypes>> {
125            self.events
126                .read()
127                .await
128                .iter()
129                .flat_map(|event| {
130                    let EventType::Decide { leaf_chain, .. } = &event.event else {
131                        panic!("expected decide event, got {event:?}");
132                    };
133                    leaf_chain.iter().cloned().rev()
134                })
135                .collect::<Vec<_>>()
136        }
137    }

142            self.events.write().await.push(event.clone());
143            Ok(())
144        }

254    pub async fn test_store_drb_input<P: TestablePersistence>(_p: PhantomData<P>) {
255        use hotshot_types::drb::DrbInput;
256        setup_test();
257
258        let tmp = P::tmp_storage().await;
259        let storage = P::connect(&tmp).await;
260        let difficulty_level = 10;
261
262        // Initially, there is no saved info.
263        if storage.load_drb_input(10).await.is_ok() {
264            panic!("unexpected nonempty drb_input");
265        }
266
267        let drb_input_1 = DrbInput {
268            epoch: 10,
269            iteration: 10,
270            value: [0u8; 32],
271            difficulty_level,
272        };
273
274        let drb_input_2 = DrbInput {
275            epoch: 10,
276            iteration: 20,
277            value: [0u8; 32],
278            difficulty_level,
279        };
280
281        let drb_input_3 = DrbInput {
282            epoch: 10,
283            iteration: 30,
284            value: [0u8; 32],
285            difficulty_level,
286        };
287
288        let _ = storage.store_drb_input(drb_input_1.clone()).await;
289
290        assert_eq!(storage.load_drb_input(10).await.unwrap(), drb_input_1);
291
292        let _ = storage.store_drb_input(drb_input_3.clone()).await;
293
294        // check that the drb input is overwritten
295        assert_eq!(storage.load_drb_input(10).await.unwrap(), drb_input_3);
296
297        let _ = storage.store_drb_input(drb_input_2.clone()).await;
298
299        // check that the drb input is not overwritten by the older value
300        assert_eq!(storage.load_drb_input(10).await.unwrap(), drb_input_3);
301    }

304    pub async fn test_epoch_info<P: TestablePersistence>(_p: PhantomData<P>) {
305        setup_test();
306        let tmp = P::tmp_storage().await;
307        let storage = P::connect(&tmp).await;
308
309        // Initially, there is no saved info.
310        assert_eq!(storage.load_start_epoch_info().await.unwrap(), Vec::new());
311
312        // Store a drb result.
313        storage
314            .store_drb_result(EpochNumber::new(1), [1; 32])
315            .await
316            .unwrap();
317        assert_eq!(
318            storage.load_start_epoch_info().await.unwrap(),
319            vec![InitializerEpochInfo::<SeqTypes> {
320                epoch: EpochNumber::new(1),
321                drb_result: [1; 32],
322                block_header: None,
323            }]
324        );
325
326        // Store a second DRB result
327        storage
328            .store_drb_result(EpochNumber::new(2), [3; 32])
329            .await
330            .unwrap();
331        assert_eq!(
332            storage.load_start_epoch_info().await.unwrap(),
333            vec![
334                InitializerEpochInfo::<SeqTypes> {
335                    epoch: EpochNumber::new(1),
336                    drb_result: [1; 32],
337                    block_header: None,
338                },
339                InitializerEpochInfo::<SeqTypes> {
340                    epoch: EpochNumber::new(2),
341                    drb_result: [3; 32],
342                    block_header: None,
343                }
344            ]
345        );
346
347        // Make a header
348        let instance_state = NodeState::mock();
349        let validated_state = hotshot_types::traits::ValidatedState::genesis(&instance_state).0;
350        let leaf: Leaf2 = Leaf::genesis::<MockVersions>(&validated_state, &instance_state)
351            .await
352            .into();
353        let header = leaf.block_header().clone();
354
355        // Test storing the header
356        storage
357            .store_epoch_root(EpochNumber::new(1), header.clone())
358            .await
359            .unwrap();
360        assert_eq!(
361            storage.load_start_epoch_info().await.unwrap(),
362            vec![
363                InitializerEpochInfo::<SeqTypes> {
364                    epoch: EpochNumber::new(1),
365                    drb_result: [1; 32],
366                    block_header: Some(header.clone()),
367                },
368                InitializerEpochInfo::<SeqTypes> {
369                    epoch: EpochNumber::new(2),
370                    drb_result: [3; 32],
371                    block_header: None,
372                }
373            ]
374        );
375
376        // Store more than the limit
377        let total_epochs = RECENT_STAKE_TABLES_LIMIT + 10;
378        for i in 0..total_epochs {
379            let epoch = EpochNumber::new(i);
380            let drb = [i as u8; 32];
381            storage
382                .store_drb_result(epoch, drb)
383                .await
384                .unwrap_or_else(|_| panic!("Failed to store DRB result for epoch {i}"));
385        }
386
387        let results = storage.load_start_epoch_info().await.unwrap();
388
389        // Check that only the most recent RECENT_STAKE_TABLES_LIMIT epochs are returned
390        assert_eq!(
391            results.len(),
392            RECENT_STAKE_TABLES_LIMIT as usize,
393            "Should return only the most recent {RECENT_STAKE_TABLES_LIMIT} epochs",
394        );
395
396        for (i, info) in results.iter().enumerate() {
397            let expected_epoch =
398                EpochNumber::new(total_epochs - RECENT_STAKE_TABLES_LIMIT + i as u64);
399            let expected_drb = [(total_epochs - RECENT_STAKE_TABLES_LIMIT + i as u64) as u8; 32];
400            assert_eq!(info.epoch, expected_epoch, "invalid epoch at index {i}",);
401            assert_eq!(info.drb_result, expected_drb, "invalid DRB at index {i}",);
402            assert!(info.block_header.is_none(), "Expected no block header");
403        }
404    }
