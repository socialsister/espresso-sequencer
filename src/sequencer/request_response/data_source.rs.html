71/// Implement the trait that allows the [`RequestResponseProtocol`] to calculate/derive a response for a specific request
72#[async_trait]
73impl<
74        I: NodeImplementation<SeqTypes>,
75        V: Versions,
76        N: ConnectedNetwork<PubKey>,
77        P: SequencerPersistence,
78    > DataSourceTrait<Request> for DataSource<I, V, N, P>
79{
80    async fn derive_response_for(&self, request: &Request) -> Result<Response> {
81        match request {
82            Request::Accounts(height, view, accounts) => {
83                // Try to get accounts from memory first, then fall back to storage
84                if let Some(state) = self.consensus.state(ViewNumber::new(*view)).await {
85                    if let Ok(accounts) =
86                        retain_accounts(&state.fee_merkle_tree, accounts.iter().copied())
87                    {
88                        return Ok(Response::Accounts(accounts));
89                    }
90                }
91
92                // Fall back to storage
93                let (merkle_tree, leaf) = match &self.storage {
94                    Some(Storage::Sql(storage)) => storage
95                        .get_accounts(&self.node_state, *height, ViewNumber::new(*view), accounts)
96                        .await
97                        .with_context(|| "failed to get accounts from sql storage")?,
98                    Some(Storage::Fs(_)) => bail!("fs storage not supported for accounts"),
99                    _ => bail!("storage was not initialized"),
100                };
101
102                // If we successfully fetched accounts from storage, try to add them back into the in-memory
103                // state.
104                if let Err(err) = add_fee_accounts_to_state::<N, V, P>(
105                    &self.consensus.consensus(),
106                    &ViewNumber::new(*view),
107                    accounts,
108                    &merkle_tree,
109                    leaf,
110                )
111                .await
112                {
113                    tracing::warn!(?view, "Cannot update fetched account state: {err:#}");
114                }
115
116                Ok(Response::Accounts(merkle_tree))
117            },
118
119            Request::Leaf(height) => {
120                // Try to get the leaves from memory first, then fall back to storage
121                let mut leaves = self.consensus.consensus().read().await.undecided_leaves();
122                leaves.sort_by_key(|l| l.view_number());
123
124                if let Some((position, mut last_leaf)) =
125                    leaves.iter().find_position(|l| l.height() == *height)
126                {
127                    let mut leaf_chain = vec![last_leaf.clone()];
128                    for leaf in leaves.iter().skip(position + 1) {
129                        if leaf.justify_qc().view_number() == last_leaf.view_number() {
130                            leaf_chain.push(leaf.clone());
131                        } else {
132                            continue;
133                        }
134                        if leaf.view_number() == last_leaf.view_number() + 1 {
135                            // one away from decide
136                            last_leaf = leaf;
137                            break;
138                        }
139                        last_leaf = leaf;
140                    }
141
142                    // Make sure we got one more leaf to confirm the decide
143                    for leaf in leaves
144                        .iter()
145                        .skip_while(|l| l.view_number() <= last_leaf.view_number())
146                    {
147                        if leaf.justify_qc().view_number() == last_leaf.view_number() {
148                            leaf_chain.push(leaf.clone());
149                            return Ok(Response::Leaf(leaf_chain));
150                        }
151                    }
152                }
153
154                // Fall back to storage
155                let leaf_chain = match &self.storage {
156                    Some(Storage::Sql(storage)) => storage
157                        .get_leaf_chain(*height)
158                        .await
159                        .with_context(|| "failed to get leaf from sql storage")?,
160                    // TODO: Actually implement FS storage for some of these
161                    Some(Storage::Fs(_)) => bail!("fs storage not supported for leaf"),
162                    _ => bail!("storage was not initialized"),
163                };
164
165                Ok(Response::Leaf(leaf_chain))
166            },
167            Request::ChainConfig(commitment) => {
168                // Try to get the chain config from memory first, then fall back to storage
169                let chain_config_from_memory = self.consensus.decided_state().await.chain_config;
170                if chain_config_from_memory.commit() == *commitment {
171                    if let Some(chain_config) = chain_config_from_memory.resolve() {
172                        return Ok(Response::ChainConfig(chain_config));
173                    }
174                }
175
176                // Fall back to storage
177                Ok(Response::ChainConfig(match &self.storage {
178                    Some(Storage::Sql(storage)) => storage
179                        .get_chain_config(*commitment)
180                        .await
181                        .with_context(|| "failed to get chain config from sql storage")?,
182                    Some(Storage::Fs(_)) => {
183                        bail!("fs storage not supported for chain config")
184                    },
185                    _ => bail!("storage was not initialized"),
186                }))
187            },
188            Request::BlocksFrontier(height, view) => {
189                // First try to respond from memory
190                let blocks_frontier_from_memory: Option<Result<BlocksFrontier>> = self
191                    .consensus
192                    .state(ViewNumber::new(*view))
193                    .await
194                    .map(|state| {
195                        let tree = &state.block_merkle_tree;
196                        let frontier = tree.lookup(tree.num_leaves() - 1).expect_ok()?.1;
197                        Ok(frontier)
198                    });
199
200                if let Some(Ok(blocks_frontier_from_memory)) = blocks_frontier_from_memory {
201                    return Ok(Response::BlocksFrontier(blocks_frontier_from_memory));
202                } else {
203                    // If we can't get the blocks frontier from memory, fall through to storage
204                    let blocks_frontier_from_storage = match &self.storage {
205                        Some(Storage::Sql(storage)) => storage
206                            .get_frontier(&self.node_state, *height, ViewNumber::new(*view))
207                            .await
208                            .with_context(|| "failed to get blocks frontier from sql storage")?,
209                        Some(Storage::Fs(_)) => {
210                            bail!("fs storage not supported for blocks frontier")
211                        },
212                        _ => bail!("storage was not initialized"),
213                    };
214
215                    Ok(Response::BlocksFrontier(blocks_frontier_from_storage))
216                }
217            },
218            Request::RewardAccounts(height, view, accounts) => {
219                // Try to get the reward accounts from memory first, then fall back to storage
220                if let Some(state) = self.consensus.state(ViewNumber::new(*view)).await {
221                    if let Ok(reward_accounts) =
222                        retain_reward_accounts(&state.reward_merkle_tree, accounts.iter().copied())
223                    {
224                        return Ok(Response::RewardAccounts(reward_accounts));
225                    }
226                }
227
228                // Fall back to storage
229                let (merkle_tree, leaf) = match &self.storage {
230                    Some(Storage::Sql(storage)) => storage
231                        .get_reward_accounts(
232                            &self.node_state,
233                            *height,
234                            ViewNumber::new(*view),
235                            accounts,
236                        )
237                        .await
238                        .with_context(|| "failed to get accounts from sql storage")?,
239                    Some(Storage::Fs(_)) => {
240                        bail!("fs storage not supported for reward accounts")
241                    },
242                    _ => bail!("storage was not initialized"),
243                };
244
245                // If we successfully fetched accounts from storage, try to add them back into the in-memory
246                // state.
247                if let Err(err) = add_reward_accounts_to_state::<N, V, P>(
248                    &self.consensus.consensus(),
249                    &ViewNumber::new(*view),
250                    accounts,
251                    &merkle_tree,
252                    leaf,
253                )
254                .await
255                {
256                    tracing::warn!(?view, "Cannot update fetched account state: {err:#}");
257                }
258
259                Ok(Response::RewardAccounts(merkle_tree))
260            },
261
262            Request::VidShare(block_number, _request_id) => {
263                // Load the VID share from storage
264                let vid_share = match &self.storage {
265                    Some(Storage::Sql(storage)) => storage
266                        .get_vid_share::<SeqTypes>(BlockId::Number(*block_number as usize))
267                        .await
268                        .with_context(|| "failed to get vid share from sql storage")?,
269                    Some(Storage::Fs(storage)) => {
270                        // Open a read transaction
271                        let mut transaction = storage
272                            .read()
273                            .await
274                            .with_context(|| "failed to open fs storage transaction")?;
275
276                        // Get the VID share
277                        transaction
278                            .vid_share(BlockId::Number(*block_number as usize))
279                            .await
280                            .with_context(|| "failed to get vid share from fs storage")?
281                    },
282                    _ => bail!("storage was not initialized"),
283                };
284
285                Ok(Response::VidShare(vid_share))
286            },
287        }
288    }
289}

291/// Get a partial snapshot of the given reward state, which contains only the specified accounts.
292///
293/// Fails if one of the requested accounts is not represented in the original `state`.
294pub fn retain_reward_accounts(
295    state: &RewardMerkleTree,
296    accounts: impl IntoIterator<Item = RewardAccount>,
297) -> anyhow::Result<RewardMerkleTree> {
298    let mut snapshot = RewardMerkleTree::from_commitment(state.commitment());
299    for account in accounts {
300        match state.universal_lookup(account) {
301            LookupResult::Ok(elem, proof) => {
302                // This remember cannot fail, since we just constructed a valid proof, and are
303                // remembering into a tree with the same commitment.
304                snapshot.remember(account, *elem, proof).unwrap();
305            },
306            LookupResult::NotFound(proof) => {
307                // Likewise this cannot fail.
308                snapshot.non_membership_remember(account, proof).unwrap()
309            },
310            LookupResult::NotInMemory => {
311                bail!("missing account {account}");
312            },
313        }
314    }
315
316    Ok(snapshot)
